261      public void EscribirEnFichero(String correo, String contrasena) throws IOException {
262          OutputStreamWriter osw = new OutputStreamWriter(openFileOutput("user.txt", Activity.MODE_PRIVATE));
263          osw.write(correo + "\n" + contrasena);
264          osw.flush();
265          osw.close();

269      @Override
270      protected void onActivityResult(int requestCode, int resultCode, @Nullable Intent data) {
271          super.onActivityResult(requestCode, resultCode, data);
272          if (requestCode == imagen_request && resultCode == RESULT_OK && data != null) {
273              Uri ruta = data.getData();
274              try {
275                  bitmap = MediaStore.Images.Media.getBitmap(getContentResolver(), ruta);
276                  ibUsuario.setImageBitmap(bitmap);
277                  ByteArrayOutputStream baos = new ByteArrayOutputStream();
278                  bitmap.compress(Bitmap.CompressFormat.JPEG, 50, baos);
279                  imagenByte = baos.toByteArray();
280  
281              } catch (IOException e) {
282                  e.printStackTrace();
283              }
284          }
