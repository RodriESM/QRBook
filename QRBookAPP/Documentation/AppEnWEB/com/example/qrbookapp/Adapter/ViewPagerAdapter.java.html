11   import java.util.ArrayList;

14   public class ViewPagerAdapter extends FragmentPagerAdapter {
15   
16       private List<Fragment> fragmentList = new ArrayList<>();
17       private CharSequence[] Titles;

19       private Fragment_ListaLibros_Usuario FLU = new Fragment_ListaLibros_Usuario();
20   
21       //Pasamos el array de los títulos.
22       public ViewPagerAdapter(FragmentManager fm, CharSequence[] mTitles) {
23           super(fm);
24           this.Titles = mTitles;
25       }
26   
27       //Modificar para que dependiendo de la posición del tab muestre un fragment u otro
28       @NonNull
29       @Override
30       public Fragment getItem(int position) {
31           //Dependiendo del de la tabulación abrirá uno u otro y por defecto la lista de libros.
32           if (position == 1) {
33               return FLU;
34           }
35           return FL;
36       }
37   
38       @Override
39       public int getCount() {
40           return fragmentList.size();
41       }
42   
43       public void addFragment(Fragment fragment) {
44           fragmentList.add(fragment);
45       }
46   
47       //Ponemos el título que le pertenece a cada tab.
48       @Override
49       public CharSequence getPageTitle(int position) {
50           return Titles[position];
51       }
