150          @Override
151          //Método al que s ele pasa una clase Detector con una colección de items detectados.
152          public void receiveDetections(Detector.Detections<Barcode> detections) {
153  
154              Bundle bundle = getIntent().getExtras();
155              assert bundle != null;
156              String isbn = bundle.getString("escanerisbn");
157  
158              Connection connection = ConnectionClass.con;
159  
160              final String[]  datos= fileList();
161              final String nombreFicheroRecordatorio = "user.txt";
162  
163  
164              if (accesoFichero.archivoExisteEntreFicheros(datos, nombreFicheroRecordatorio)) {
165  
166                  try {
167                      InputStreamReader isr = new InputStreamReader(openFileInput(nombreFicheroRecordatorio));
168                      BufferedReader br = new BufferedReader(isr);
169                      String linea = br.readLine();
170  
171                      //Introducimos los datos en un array recorriendo cada linea del fichero, en la primera linea tendrá el usuario y en la segunda la contraseña
172                      while (linea != null) {
173                          contenidoFicheroRecordado.add(linea);
174                          linea = br.readLine();
175                      }
176                  } catch (IOException e) {
177                      e.printStackTrace();
178                  } finally {
179                      correo = contenidoFicheroRecordado.get(0);
180                  }
181              }
182  
183  
184              if (detections != null && detections.getDetectedItems().size() != 0) {
185                  SparseArray<Barcode> qrCodes = detections.getDetectedItems();
186                  Barcode code = qrCodes.valueAt(0);
187  
188                  if ((code.displayValue.contains("http") || code.displayValue.contains("https")) && !code.displayValue.contentEquals(lastUrl)) {
189                      lastUrl = code.displayValue;
190                      ResultSet rs;
191                      ResultSet rsgeneral;
192  
193                      try {
194                          //Revisamos todos los qr de la base de datos.
195                          rsgeneral = connection.createStatement().executeQuery("select * from QR where URL like'" + code.displayValue + "' and ISBN like '" + isbn + "'");
196  
197                          //Si encuentra el QR, lo muestra.
198                          if (rsgeneral.next()) {
199  
200                              Intent i = new Intent(Escaner.this, EscanerVista.class);
201                              i.putExtra("url", code.displayValue);
202                              i.setFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP);
203                              i.setFlags(Intent.FLAG_ACTIVITY_SINGLE_TOP);
204                              startActivity(i);
205  
206  
207                          } else {
208  
209                              try {
210                                  //Miramos que el qr no esté registrado para no crear duplicados en nuestro contenedor de qr
211                                  rs = connection.createStatement().executeQuery("SELECT * FROM USUARIOQR where URL like '" + code.displayValue + "' and CORREO like '" + correo + "' and ISBN like '" + isbn + "'");
212  
213                                  //Recorremos todos lo libros que tenemos en la base de datos y los introducimos en el array
214                                  if (rs.next()) {
215                                      Intent i = new Intent(Escaner.this, EscanerVista.class);
216                                      i.putExtra("url", code.displayValue);
217                                      i.setFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP);
218                                      i.setFlags(Intent.FLAG_ACTIVITY_SINGLE_TOP);
219                                      startActivity(i);
220  
221                                  } else {
222  //
223  //                                   /*
224  //                                    Si queremos abrirla en el navegador por defecto usariamos este código o Script
225  //                                    Uri uri=Uri.parse(code.displayValue);
226  //                                    Intent i=new Intent(Intent.ACTION_VIEW,uri);*/
227  //                                    //Al querer abrirlo en otra actividad, pasamos los datos a la otra actividad de esta forma
228  //
229  //
230  //                                    PreparedStatement ps = connection.prepareStatement("INSERT INTO USUARIOQR(CORREO,URL,ISBN) values(?,?,?)");
231  //                                    ps.setString(1, correo);
232  //                                    ps.setString(2, code.displayValue);
233  //                                    ps.setString(3,isbn);
234  //                                    ps.executeUpdate();
235  
236                                      Intent i = new Intent(Escaner.this, AnadirQrUsuario.class);
237                                      i.putExtra("url", code.displayValue);
238                                      i.putExtra("correo", correo);
239                                      i.putExtra("isbn", isbn);
240                                      i.setFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP);
241                                      i.setFlags(Intent.FLAG_ACTIVITY_SINGLE_TOP);
242                                      startActivity(i);
243  
244                                  }
245  
246                              } catch (SQLException e) {
247                                  e.printStackTrace();
248                              }
249                          }
250                      } catch (SQLException e) {
251                          e.printStackTrace();
252                      }
253  //Revisar si nos hiciera falta...
254                  }

258              }
