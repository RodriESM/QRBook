1    package com.example.qrbookapp;

3    import android.Manifest;
4    import android.annotation.SuppressLint;
5    import android.content.pm.PackageManager;
6    import android.os.Bundle;

8    import android.widget.Toast;
9    
10   import androidx.appcompat.app.AppCompatActivity;
11   import androidx.core.app.ActivityCompat;
12   import androidx.core.content.ContextCompat;
13   
14   import com.github.barteksc.pdfviewer.PDFView;
15   
16   public class PDF extends AppCompatActivity {
17   
18       @SuppressLint("SdCardPath")
19       @Override
20       protected void onCreate(Bundle savedInstanceState) {
21           super.onCreate(savedInstanceState);
22           setContentView(R.layout.activity_p_d_f);
23   
24           Bundle b = getIntent().getExtras();
25           assert b != null;
26           String isbn = b.getString("isbn");
27           String correo = b.getString("correo");
28   
29   // Here, thisActivity is the current activity
30           if (ContextCompat.checkSelfPermission(PDF.this,
31                   Manifest.permission.READ_EXTERNAL_STORAGE)
32                   != PackageManager.PERMISSION_GRANTED) {
33   
34               Toast.makeText(this, "Para poder leer el libro hay que aceptar los permisos", Toast.LENGTH_LONG).show();
35   
36               if (ActivityCompat.shouldShowRequestPermissionRationale(PDF.this,
37                       Manifest.permission.READ_EXTERNAL_STORAGE)) {
38                   // Show an explanation to the user *asynchronously* -- don't block
39                   // this thread waiting for the user's response! After the user
40                   // sees the explanation, try again to request the permission.
41               } else {
42                   // No explanation needed; request the permission
43                   ActivityCompat.requestPermissions(PDF.this,
44                           new String[]{Manifest.permission.READ_EXTERNAL_STORAGE}, PackageManager.PERMISSION_GRANTED);
45   
46                   // MY_PERMISSIONS_REQUEST_READ_CONTACTS is an
47                   // app-defined int constant. The callback method gets the
48                   // result of the request.
49               }
50           } else {
51               //PDF View
52               PDFView pdfView = findViewById(R.id.pdfView);
53               pdfView.fromFile(getDatabasePath("/sdcard/Documents/" + isbn + correo + ".pdf"))
54                       .enableSwipe(true) // allows to block changing pages using swipe
55                       .swipeHorizontal(false)
56                       .enableDoubletap(true)
57                       .defaultPage(0)
58                       .enableAnnotationRendering(false) // render annotations (such as comments, colors or forms)
59                       .password(null)
60                       .scrollHandle(null)
61                       .enableAntialiasing(true) // improve rendering a little bit on low-res screens
62                       // spacing between pages in dp. To define spacing color, set view background
63                       .spacing(0)
64                       .autoSpacing(false) // add dynamic spacing to fit each page on its own on the screen
65                       .fitEachPage(false) // fit each page to the view, else smaller pages are scaled relative to largest page.
66                       .pageSnap(false) // snap pages to screen boundaries
67                       .pageFling(false) // make a fling change only a single page like ViewPager
68                       .nightMode(false)// toggle night mode
69                       .load();
70           }
71       }
72   
73       @Override
74       public boolean onKeyDown(int keyCode, KeyEvent event) {
75           if ((keyCode == KeyEvent.KEYCODE_BACK)) {
76               super.finish();
77           }
78           return super.onKeyDown(keyCode, event);
79       }
80   }
